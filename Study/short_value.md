#include <stdio.h>
main()
{
	struct{
		short a:8,
		      data:8;
	}a;
	a.data = 0x00;
	int i;
	for (i = 0; i < 258; i++) {
		a.data += 1;
		printf("%#02hhx  %d\n",a.data, a.data);
	}
}
